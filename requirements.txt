# Pip-based install for environments where conda isn't available
# (e.g. the AWS deploy target — see dev_docs/DEPLOY.md).  This file
# MUST stay in sync with environment.yaml's pip: section.
#
# Scope: everything needed to run mcp_server.py AND the build/
# reconcile CLIs.  The heavy vision + OCR extras (docling,
# ocrmypdf, qwen-vl-utils, accelerate) are included for parity
# with the conda env but can be stripped on a server-only host
# to save ~2 GB of disk.

# Web, data, config
requests
python-dotenv
pydantic
pyyaml
numpy
pillow
matplotlib
rapidfuzz>=2.0
lxml

# MCP server + transport deps.  uvicorn is also a transitive dep of
# mcp today, but we pin it explicitly because mcp_server.py imports it
# by name for the SSE transport.
mcp
uvicorn

# Vector index + embedding backend.  torch + transformers come in via
# sentence-transformers, but vision.py and embeddings.py import them
# directly, so we declare them to avoid relying on a transitive pin.
lancedb
sentence-transformers
torch
transformers

# LLM clients used by MCP tools
anthropic

# Pipeline-side (needed for process_corpus.py / figures.py, not
# strictly required by the MCP server alone)
docling
ocrmypdf
pymupdf
langdetect
pytesseract
qwen-vl-utils
accelerate

# Notebooks / interactive dev
ipykernel

# Test runner
pytest
